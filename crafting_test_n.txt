     1	<?php
     2
     3	namespace Tests\Feature;
     4
     5	use App\Models\Resource;
     6	use App\Models\User;
     7	use Illuminate\Foundation\Testing\RefreshDatabase;
     8	use Tests\TestCase;
     9
    10	class CraftingTest extends TestCase
    11	{
    12	    use RefreshDatabase;
    13
    14	    public function test_user_can_craft_weapon_if_has_resources(): void
    15	    {
    16	        $user = User::create(['username' => 'blacksmith', 'password' => 'p']);
    17	        Resource::create(['user_id' => $user->id, 'type' => 'metal', 'amount' => 20]);
    18	        Resource::create(['user_id' => $user->id, 'type' => 'energy', 'amount' => 10]);
    19
    20	        $response = $this->actingAs($user)->post('/craft', [
    21	            'weapon_type' => 'Iron Sword',
    22	        ]);
    23
    24	        $response->assertStatus(302);
    25	        $this->assertDatabaseHas('weapons', ['user_id' => $user->id, 'type' => 'Iron Sword']);
    26	        $this->assertEquals(10, Resource::where('user_id', $user->id)->where('type' => 'metal')->first()->amount);
    27	        $this->assertEquals(5, Resource::where('user_id', $user->id)->where('type' => 'energy')->first()->amount);
    28	    }
    29
    30	    public function test_user_cannot_craft_without_resources(): void
    31	    {
    32	        $user = User::create(['username' => 'poor_smith', 'password' => 'p']);
    33	        Resource::create(['user_id' => $user->id, 'type' => 'metal', 'amount' => 5]);
    34
    35	        $response = $this->actingAs($user)->post('/craft', [
    36	            'weapon_type' => 'Iron Sword',
    37	        ]);
    38
    39	        $response->assertSessionHasErrors(['message' => 'Insufficient resources']);
    40	    }
    41	}
